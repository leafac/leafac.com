export class PegSolitaire extends React.Component {
  constructor() {
    super();
    this.PEG = "●";
    this.SPACE = "○";
    this.PADDING = "·";
    this.INITIAL_STATE = {
      board: this.parse(`
        · · ● ● ● · ·
        · · ● ● ● · ·
        ● ● ● ● ● ● ●
        ● ● ● ○ ● ● ●
        ● ● ● ● ● ● ●
        · · ● ● ● · ·
        · · ● ● ● · ·
      `),
      selected: null,
    };
    this.SIZE = this.INITIAL_STATE.board[0].length;
    this.RENDERABLES = [this.PEG, this.SPACE];
    this.state = this.INITIAL_STATE;
    this.reset = this.reset.bind(this);
    this.cheat = this.cheat.bind(this);
  }
  parse(string) {
    return string
      .trim()
      .split("\n")
      .map((line) => line.trim().split(" "));
  }
  neighbors({ x, y }) {
    return [
      { x, y: y - 2 },
      { x: x + 2, y },
      { x, y: y + 2 },
      { x: x - 2, y },
    ].filter(({ x, y }) => this.RENDERABLES.includes(this.state.board[y]?.[x]));
  }
  halfway(coordinate1, coordinate2) {
    return {
      x: (coordinate1.x + coordinate2.x) / 2,
      y: (coordinate1.y + coordinate2.y) / 2,
    };
  }
  same(coordinate1, coordinate2) {
    return JSON.stringify(coordinate1) === JSON.stringify(coordinate2);
  }
  createOnClick(current) {
    const { board, selected } = this.state;
    if (!this.RENDERABLES.includes(board[current.y][current.x])) return null;
    if (
      selected === null &&
      board[current.y][current.x] === this.PEG &&
      this.neighbors(current).some((neighbor) => {
        const halfway = this.halfway(current, neighbor);
        return (
          board[neighbor.y][neighbor.x] === this.SPACE &&
          board[halfway.y][halfway.x] === this.PEG
        );
      })
    )
      return () => {
        this.setState({ selected: current });
      };
    if (this.same(current, selected))
      return () => {
        this.setState({ selected: null });
      };
    if (
      selected !== null &&
      board[current.y][current.x] === this.SPACE &&
      this.neighbors(current).some((neighbor) => {
        const halfway = this.halfway(current, neighbor);
        return (
          this.same(neighbor, selected) &&
          board[halfway.y][halfway.x] === this.PEG
        );
      })
    )
      return () => {
        const newBoard = JSON.parse(JSON.stringify(board));
        const halfway = this.halfway(current, selected);
        newBoard[current.y][current.x] = this.PEG;
        newBoard[selected.y][selected.x] = this.SPACE;
        newBoard[halfway.y][halfway.x] = this.SPACE;
        this.setState({ board: newBoard, selected: null });
      };
    return null;
  }
  isOver() {
    return !this.state.board.some((row, y) =>
      row.some((cell, x) => Boolean(this.createOnClick({ x, y })))
    );
  }
  isWin() {
    return (
      this.state.board.flat().filter((cell) => cell === this.PEG).length === 1
    );
  }
  reset() {
    this.setState(this.INITIAL_STATE);
  }
  cheat() {
    this.setState({
      board: this.parse(`
      · · ○ ○ ○ · ·
      · · ○ ○ ○ · ·
      ○ ○ ○ ○ ○ ○ ○
      ○ ● ○ ● ● ○ ○
      ○ ○ ○ ○ ○ ○ ○
      · · ○ ○ ○ · ·
      · · ○ ○ ○ · ·
    `),
      selected: null,
    });
  }
  render() {
    return (
      <table
        style={{
          display: "inline-block",
          textAlign: "center",
          borderCollapse: "collapse",
        }}
      >
        <thead>
          <tr>
            <td colspan={this.SIZE}>
              {this.isOver() ? (this.isWin() ? "🎉" : "🙁") : " "}
            </td>
          </tr>
        </thead>
        <tbody style={{ lineHeight: "1em" }}>
          {this.state.board.map((row, y) => (
            <tr key={y}>
              {row.map((cell, x) => {
                const onClick = this.createOnClick({ x, y });
                return (
                  <td key={x} style={{ padding: 0 }}>
                    {this.RENDERABLES.includes(cell) ? (
                      <button
                        onClick={onClick}
                        disabled={!Boolean(onClick)}
                        style={{ margin: 0, fontSize: "1em" }}
                      >
                        <code>{cell}</code>
                      </button>
                    ) : null}
                  </td>
                );
              })}
            </tr>
          ))}
        </tbody>
        <tfoot>
          <tr>
            <td colspan={this.SIZE}>
              <button onClick={this.reset}>Reset</button>
              <button onClick={this.cheat}>Cheat</button>
            </td>
          </tr>
        </tfoot>
      </table>
    );
  }
}
